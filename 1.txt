using namespace std;

    setlocale(LC_ALL, "Russian"); 
    int month; 
     
    // Вывод меню 
    cout << "[ + ] Месяца года\n";

    cout << "[ 2 ] Февраль\n";

    cout << "[ 12 ] Декабрь\n"; 
    cout << "\n"; 
     
    // Ввод номера месяца 
    cout << "[ + ] Выберите месяц: ";

    switch(month) {

            break; 
        case 2:

            break; 
        case 3: 
            cout << "[ + ] Выбран месяц \"Март\"\n"; 
            break; 
        case 4: 
            cout << "[ + ] Выбран месяц \"Апрель\"\n"; 
            break; 
        case 5: 
            cout << "[ + ] Выбран месяц \"Май\"\n"; 
            break; 
        case 6: 
            cout << "[ + ] Выбран месяц \"Июнь\"\n"; 
            break; 
        case 7: 
            cout << "[ + ] Выбран месяц \"Июль\"\n"; 
            break; 
        case 8: 
            cout << "[ + ] Выбран месяц \"Август\"\n"; 
            break; 
        case 9: 
            cout << "[ + ] Выбран месяц \"Сентябрь\"\n"; 
            break;

            cout << "[ + ] Выбран месяц \"Октябрь\"\n";
